3 - Apresente qual será a saída dos programas abaixo:

a)  #include <stdio.h>
    int main() {
        int a;
        int *p;
        p = &a;
        a = 0;
        *p = 2;
        printf("%d", a);
        return 0;
    }

//resposta: 2

b)  #include <stdio.h>

    }

e)  #include <stdio.h>
    #include <conio.h>
    int main () {
        int valores[3][3] = {{1, 2, 3}, {4, 5, 6}, {7, 8, 9}};
        int i, j;
        int *iptr;
        iptr = &valores[0][0];
        for (i = 0; i < 3; i++) {
            for (j = 0; j < 3; j++) {
                printf ("%d ", *iptr);
                iptr++;
            }
            printf ("\n");
        }
        getch();
    }

//resposta:
1 2 3
4 5 6
7 8 9
